using UnityEngine;
using System.Collections;
using System.Collections.Generic;
using UnityEngine.UI;

public class SGUCallbackEntityObject : MonoBehaviour, SGUCallbackEntity {

    void Start () {
        SGU.initialize(this);   // SGU class initialization
        // call Init interface
        SGU.Init("com.sg.util.SGUtilUnityGameServer", false);
    }

    void Update()
    {
    }

    public void OnInit(SGUStateChangeReason reason)
    {
        if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_SUCCESS)
        {
            // initialization succeeded, show a button 'login' to login
        }
        else
        {
            // initialization failed, should terminate program
        }
    }

    public void OnLogin(SGUOp action, SGUStateChangeReason reason)
    {
        if (action == SGUOp.OP_LOGIN)
        {
            if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_SUCCESS)
            {
                // channel login succeeded, wait
            }
            else if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_FAILURE)
            {
                // channel login failed, should not continue
            }
        }
        else if (action == SGUOp.OP_AUTH)
        {
            if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_SUCCESS)
            {
                // channel login confirmation succeeded, wait
            }
            else if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_FAILURE)
            {
                // channel login confirmation failed, should not continue
            }
        }
        else if (action == SGUOp.OP_LOGINGAME)
        {
            if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_SUCCESS)
            {
                // game server login succeeded, show game screen
            }
            else if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_FAILURE)
            {
                // game server login failed, should not continue
            }
        }
    }

    public void OnPay(SGUStateChangeReason reason)
    {
        if (reason == SGUStateChangeReason.STATE_CHANGE_REASON_OP_SUCCESS)

        }

    }

    public void OnOrderInfoFound(SGUOrder order)

    }

    public void OnDebugInfo(string info)

    }
